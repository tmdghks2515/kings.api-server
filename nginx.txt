# ---- HTTP(80): ACME 인증 응답 + HTTPS 리다이렉트 ----
server {
  listen 80;
  server_name graetdae.daneyo.com;

  # Let's Encrypt webroot
  location /.well-known/acme-challenge/ {
    root /var/www/certbot;
  }

  # 나머지는 전부 HTTPS로 리다이렉트
  location / {
    return 301 https://$host$request_uri;
  }
}

# ---- HTTPS(443): 운영 트래픽 ----
server {
  listen 443 ssl;
  http2 on;
  server_name graetdae.daneyo.com;

  # ----- TLS (Let's Encrypt) -----
  ssl_certificate     /etc/letsencrypt/live/graetdae.daneyo.com/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/graetdae.daneyo.com/privkey.pem;
  ssl_session_timeout 1d;
  ssl_session_cache   shared:SSL:10m;
  ssl_protocols       TLSv1.2 TLSv1.3;
  ssl_prefer_server_ciphers on;

  # ----- 보안/운영 헤더 (필요에 따라 조정) -----
  add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;
  add_header X-Content-Type-Options nosniff always;
  add_header X-Frame-Options SAMEORIGIN always;
  add_header Referrer-Policy strict-origin-when-cross-origin always;

  # ----- 공통 프록시 설정 -----
  client_max_body_size 20m;

  # 타임아웃 (SSE/긴 응답 대비)
  proxy_read_timeout    300s;
  proxy_send_timeout    300s;
  proxy_connect_timeout 60s;

  # 공통 헤더
  proxy_set_header Host              $host;
  proxy_set_header X-Real-IP         $remote_addr;
  proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
  proxy_set_header X-Forwarded-Proto https;
  proxy_set_header X-Forwarded-Host  $host;

  # ----- Web(Next.js: web:3000) -----
  # 정적/SSR 모두 루트로 서비스
  location / {
    proxy_pass http://bo-ui:3001;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
  }

  # ----- API 라우팅 -----
  # 주의: location 경로 끝에 / 가 있으니 proxy_pass 도 / 로 맞춤(경로 보존)
  location /api/ { proxy_pass http://api-server:8080; }
}
